<div class="row">
	<div class="col m4 s12">
		<mp:button 
			action="#{'#{toastBean.showToast}'}" 
			value="Show Server Toast" 
			size="large" />
	</div>
	<div class="col m4 s12">
		<mp:button 
			type="button" 
			onclick="MaterialPrime.toast('Hi From the client',1000,true);" 
			value="Show Client Toast" 
			size="large" />
	</div>
	<div class="col m4 s12">
		<mp:button 
			type="button" 
			onclick="PF('customToast').show(5000);" 
			value="Show Custom Toast" 
			size="large" />
	</div>
</div>
<mp:toast widgetVar="customToast">
	<f:facet name="content">
		<h:outputText value="This is a Custom Toast!"/>
		<br/>
		<a onclick="MaterialPrime.toast('Another Client Toast!',1000,true);">Click Me!</a>
	</f:facet>
</mp:toast>



@ManagedBean
@NoneScoped
public class ToastBean implements Serializable {

	private static final long serialVersionUID = 1L;
	
	public void showToast(){
		ToastService.getInstance().newToast("Hi From the server!");
	}

}
