		<mp:button action="#{'#{toastBean.showToast}'}" value="Show Toast" size="large" />
	</div>
	<div class="col m3 s6">Client</div>
	<div class="col m3 s6">
		<mp:button type="button" 
			onclick="MaterialPrime.toast('Hi From the client',1000,true);" 
			value="Show Toast" size="large" />
	</div>
</div>



@ManagedBean
@NoneScoped
public class ToastBean implements Serializable {

	private static final long serialVersionUID = 1L;
	
	public void showToast(){
		ToastService.getInstance().newToast("Hi From the server!");
	}

}
